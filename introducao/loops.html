Javascript

Abra o console no devtools apertando F12 para visualizar os logs. 
            
    //Loop for
    for (let contador = 1; contador <= 5; contador++) {
        console.log(contador)
    }
    //Loop While
    var contador = 1
    while (contador <= 5) {
        console.log(contador)
        contador++
    }

    //loop do While
    var contador = 1
    do {
        console.log(contador)
        contador++
    } while (contador <= 5)
